99 bool ISO15693CompareUid(uint8_t *Uid1, uint8_t *Uid2) {
100  if ((Uid1[0] == Uid2[7])
101  && (Uid1[1] == Uid2[6])
102  && (Uid1[2] == Uid2[5])
103  && (Uid1[3] == Uid2[4])
104  && (Uid1[4] == Uid2[3])
105  && (Uid1[5] == Uid2[2])
106  && (Uid1[6] == Uid2[1])
107  && (Uid1[7] == Uid2[0])) {
108  return true;
109  } else {
110  return false;
111  }
112 }
