174 #define CryptoMemoryXOR(inputBuf, destBuf, bufSize) ({ \
175  uint8_t *in = (uint8_t *) inputBuf; \
176  uint8_t *out = (uint8_t *) destBuf; \
177  uint16_t count = (uint16_t) bufSize; \
